165  Time rto = m_rtt->RetransmitTimeout();

167  // NS_LOG_DEBUG ( "Current RTO: " << rto.ToDouble ( Time::S ) << "s" );

172  m_seqTimeouts.get<i_timestamp>().begin();
173 
174  // timeout expired?
175  if ( entry->time + rto <= now ) {
176 
177  uint32_t seqNo = entry->seq;
178  m_seqTimeouts.get<i_timestamp>().erase( entry );
179  OnTimeout( seqNo );
180 
181  } else {
182 
183  break; // nothing else to do. All later packets need not be retransmitted
184  }
185  }

187  m_retxEvent = Simulator::Schedule( m_retxTimer, &ConsumerQos::CheckRetxTimeout, this );
